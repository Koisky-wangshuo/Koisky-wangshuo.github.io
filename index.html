DDL-数据定义语言

DDL-数据库操作

查询数据库

show databases;

#显示指定名称的数据库的创建的SQL指令
show create database db_test

$C$

创建数据库

create database <dbName>
#eg: create database db_test;

#创建数据库，当指定名称的数据库不存在时执行创建
create database if not exists <databaseName>;
# eg create database if not exists db_test;

#在创建数据库的同时指定数据库的字符集，数据存储在数据库中采用的编码格式： utf8
create database <databaseName> character set utf8;
#eg: create database db_ws character set utf8

修改数据库

#修改数据库的字符集
alter database <dbName> character set utf8;
alter database <dbName> character set = utf8;**

删除数据库

删除数据库会删除数据库中所有的数据库以及数据表中的数据

drop database <dbName>;
#eg: drop database db_test;

drop database if exists <dbName>;

使用/切换数据库

use <dbName>

DDL-数据表操作

创建数据表

数据表实际上就是一个二维的表格，一个表格是由多列组成，表格中每一列称之为表格的字段

create table students(
    stu_num char(8) not null unique, 
    stu_name varchar(20) not null, 
  stu_gender char(2) not null,
  stu_age int not null,
  stu_tel char(11) not null unique,
  stu_qq varchar(11) unique
)

#char(8) 不可变字符串
#varchar 可变字符串

查询数据表

show tables;

$ a^2 + b^2 = c^2 $

查询表结构

desc tables;

删除数据表

百度 

drop table <tableName>;
#eg: drop table students;

drop table if exists <tableName>;
eg: #drop table if exists students;

修改数据表

(alter table <tableName> ...)

#修改表名
alter table <tableName_old> rename <tableName_new>;
#eg: alter table <tableName> rename stus;

#数据表也是有字符集，默认字符集与数据库相同
alter table <tableName> character set utf8;

#添加字段(添加字段,也叫添加列)
alter table <tableName> add <columnName> varchar(200);

#修改字段（列）的列名和类型
alter table <tableName> change <oldcolumnName> <newcolumnName> <type>
#eg: alter table stus change <oldcolumnName> <newcolumnName> <type>

#只修改字段的类型
alter table <tableName> modify <columnName> varchar(400);

#删除字段
alter table <tableName> drop <columnName>;

alter table stus add stu_remark varchar(200);
